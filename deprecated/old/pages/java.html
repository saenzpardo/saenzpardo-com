import java.util.Scanner;

public class AuthoringAssistant {
   
   public static void printMenu() { // printMenu method       
	       System.out.println("MENU");
	       System.out.println("c - Number of non-whitespace characters");       
	       System.out.println("w - Number of words");
	       System.out.println("f - Find text");
	       System.out.println("r - Replace all !'s");
	       System.out.println("s - Shorten spaces");
	       System.out.println("q - Quit");  
	       System.out.println();
	       System.out.println("Choose an option:");
	   }// printMenu method
	   
	   // get number of characters that are not whitespace
	   public static int getNumOfNonWSCharacters(String inputString){ 
	      int total;	      
	      // remove spaces and count total characters
	      total = inputString.replace(" ", "").length();	    	      	  
	      return total;	    
	   }// end getNumOfNonWSCharacters method
	   
	   public static int getNumOfWords(String inputString) {// get number of words
		  int count = 0;
		  // store string into an array using space as delimiter
		  String myString[] = inputString.split("\\s+");
		  for (int i = 0; i < myString.length; ++i) {
			  count = i + 1;
		  }// end for loop
			  return count;
	   }// end getNumOfWords method
	   	   
	   public static int findText(String searchWords, String inputString) {// finally works with phrase!
		   // variables
		   int count = 0;
		   int found = 0;
		   
		   while ((found = inputString.indexOf(searchWords)) != -1 ){
			  inputString = inputString.substring(found + searchWords.length());
			  ++count;
		   }// end while loop
		   return count;
	   }// end findText method
	   
	   public static String replaceExclamation(String inputStr){// replace explanation points method		   
		   return inputStr.replaceAll("!", ".");	      
	   }// end replaceExclamation points method
	   
	   public static String shortenSpace(String inputStr){// shorten space
	      String inputString = inputStr.replaceAll("\\s+", " ");
	      return inputString;
	   }// end shortenSpace method
	   
	   public static void main(String[] args) {
		  // initialize scanner
	      Scanner scnr = new Scanner(System.in);
	            
	      // declare variables
	      String inputStr;
	      String search = " ";
	      char inputChar;
	      boolean active = true;
	      
	      // get user input
	      System.out.println("Enter a sample text:");
	      System.out.println();
	      inputStr = scnr.nextLine();
	      
	      // display user input
	      System.out.println("You entered: " + inputStr);
	      System.out.println();
	      
	      // initialize while block to keep program running
	      while (active) {
	         
	      printMenu(); // call menu method
	      inputChar = scnr.next().charAt(0); // take user input for menu
	      
	      if (inputChar == 'q') {// quit
	         active = false;  
	      } else if (inputChar == 'c') {// count non-whitespace characters	
	     	  System.out.println("Number of non-whitespace characters: " + getNumOfNonWSCharacters(inputStr)); 
	    	  System.out.println();
	      } else if (inputChar == 'w') {// word counter
	          System.out.println("Number of words: " + getNumOfWords(inputStr));  
	          System.out.println();
	      } else if (inputChar == 'f') {// find text	        
	    	   System.out.println("Enter a word or phrase to be found:");
	    	 /*
	    	   I'm not sure why I needed to add the "test = scnr.nextLine();"
	    	   It seems the /n character was consuming my "search = scnr.nextLine();"
	    	 	I ended up adding the test = scnr.nextLine(); to get user input...
	    	 	I'll read more on this later.	    	   
	    	  */
	    	    String test = scnr.nextLine(); // consume /n character
	          search = scnr.nextLine(); // actual user input
	          System.out.println("\"" + search + "\"" + " instances: " + findText(search, inputStr));
	          System.out.println();
	      } else if (inputChar == 'r') {// replace exclamation point
	    	   System.out.println("Edited text: " + replaceExclamation(inputStr));
	    	  System.out.println();
	      } else if (inputChar == 's') {// shorten space
	         System.out.println("Edited text: " + shortenSpace(inputStr));	      
	         System.out.println();
	      }// end if-else-if block       
	      }// end while block  
	}// end main
}// end authoring assistant class
